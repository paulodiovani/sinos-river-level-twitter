    Search tweets and emits data for every tweet

      _search: ->
    query = @options.query
    @client.get 'search/tweets',
      q: query
      result_type: 'recent'
    , (err, tweets, response) =>
      @emit err if err?
      for tweet in tweets.statuses
        @push tweet
      @push null
